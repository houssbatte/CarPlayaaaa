chrome.alarms.create("refresh", { periodInMinutes: 1 });


   chrome.alarms.onAlarm.addListener(() => {

     chrome.tabs.query({ active: true, currentWindow: true }, (tabs) => {

       chrome.scripting.executeScript({

         target: { tabId: tabs[0].id },

         files: ['content.js']

       });

     });

   });
